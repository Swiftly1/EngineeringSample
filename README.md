# EngineeringSample

Compiler for experimental language which targets WebAssembly

# Some Code Example

	namespace Test

	public int Test(int a, int b)
	{
		if (a > b)
		{
			return a + b;
		}
		else
		{
			return a - Qwert();
		}
	}

	public int Qwert()
	{
		return 10 * 10;
	}
